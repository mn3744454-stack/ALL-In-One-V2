Policies:
    POLICY "Permission-based delete invoice items" FOR DELETE
      USING ((EXISTS ( SELECT 1
   FROM invoices i
  WHERE ((i.id = invoice_items.invoice_id) AND has_permission(auth.uid(), i.tenant_id, 'finance.invoice.edit'::text)))))
    POLICY "Permission-based insert invoice items" FOR INSERT
      WITH CHECK ((EXISTS ( SELECT 1
   FROM invoices i
  WHERE ((i.id = invoice_items.invoice_id) AND has_permission(auth.uid(), i.tenant_id, 'finance.invoice.edit'::text)))))
    POLICY "Permission-based update invoice items" FOR UPDATE
      USING ((EXISTS ( SELECT 1
   FROM invoices i
  WHERE ((i.id = invoice_items.invoice_id) AND has_permission(auth.uid(), i.tenant_id, 'finance.invoice.edit'::text)))))
    POLICY "Tenant members can view invoice items" FOR SELECT
      USING ((EXISTS ( SELECT 1
   FROM invoices i
  WHERE ((i.id = invoice_items.invoice_id) AND is_tenant_member(auth.uid(), i.tenant_id)))))
