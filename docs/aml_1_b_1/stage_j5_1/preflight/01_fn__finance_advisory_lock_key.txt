CREATE OR REPLACE FUNCTION public._finance_advisory_lock_key(p_tenant_id uuid, p_operation text, p_idempotency_key uuid)
 RETURNS bigint
 LANGUAGE sql
 IMMUTABLE
 SET search_path TO ''
AS $function$
  SELECT pg_catalog.hashtextextended(
    'fin.idem|' || COALESCE(p_tenant_id::text, '')
      || '|' || COALESCE(p_operation, '')
      || '|' || COALESCE(p_idempotency_key::text, ''),
    0::bigint);
$function$
