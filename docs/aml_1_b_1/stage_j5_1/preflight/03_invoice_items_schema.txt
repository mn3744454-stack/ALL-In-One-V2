Check constraints:
    "invoice_items_horse_source_exclusive_chk" CHECK (NOT (horse_id IS NOT NULL AND lab_horse_id IS NOT NULL))
    "invoice_items_line_gross_nonneg_ck" CHECK (line_gross_amount IS NULL OR line_gross_amount >= 0::numeric)
    "invoice_items_line_identity_ck" CHECK (line_pretax_amount IS NULL OR line_tax_amount IS NULL OR line_gross_amount IS NULL OR round(line_pretax_amount + line_tax_amount, 2) = round(line_gross_amount, 2))
    "invoice_items_line_pretax_nonneg_ck" CHECK (line_pretax_amount IS NULL OR line_pretax_amount >= 0::numeric)
    "invoice_items_line_tax_nonneg_ck" CHECK (line_tax_amount IS NULL OR line_tax_amount >= 0::numeric)
    "invoice_items_nontaxable_zero_tax_ck" CHECK (taxable_snapshot IS NULL OR line_tax_amount IS NULL OR taxable_snapshot = true OR line_tax_amount = 0::numeric)
    "invoice_items_period_valid_ck" CHECK (period_start IS NULL AND period_end IS NULL OR period_start IS NOT NULL AND period_end IS NOT NULL AND period_end >= period_start)
    "invoice_items_service_package_exclusive_chk" CHECK (NOT (service_id IS NOT NULL AND package_id IS NOT NULL))
    "invoice_items_service_source_chk" CHECK (service_source = ANY (ARRAY['tenant_services'::text, 'lab_services'::text]))
    "invoice_items_tax_rate_snapshot_range_ck" CHECK (tax_rate_snapshot IS NULL OR tax_rate_snapshot >= 0::numeric AND tax_rate_snapshot <= 100::numeric)
    "invoice_items_zero_rate_zero_tax_ck" CHECK (tax_rate_snapshot IS NULL OR line_tax_amount IS NULL OR tax_rate_snapshot <> 0::numeric OR line_tax_amount = 0::numeric)

Policies:
    POLICY "Permission-based delete invoice items" FOR DELETE
      USING ((EXISTS ( SELECT 1
   FROM invoices i
  WHERE ((i.id = invoice_items.invoice_id) AND has_permission(auth.uid(), i.tenant_id, 'finance.invoice.edit'::text)))))
    POLICY "Permission-based insert invoice items" FOR INSERT
      WITH CHECK ((EXISTS ( SELECT 1
   FROM invoices i
  WHERE ((i.id = invoice_items.invoice_id) AND has_permission(auth.uid(), i.tenant_id, 'finance.invoice.edit'::text)))))
    POLICY "Permission-based update invoice items" FOR UPDATE
      USING ((EXISTS ( SELECT 1
   FROM invoices i
  WHERE ((i.id = invoice_items.invoice_id) AND has_permission(auth.uid(), i.tenant_id, 'finance.invoice.edit'::text)))))
    POLICY "Tenant members can view invoice items" FOR SELECT
      USING ((EXISTS ( SELECT 1
   FROM invoices i
