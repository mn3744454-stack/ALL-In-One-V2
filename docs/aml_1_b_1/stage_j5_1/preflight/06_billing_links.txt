                           Table "public.billing_links"
   Column    |           Type           | Collation | Nullable |      Default      
-------------+--------------------------+-----------+----------+-------------------
 id          | uuid                     |           | not null | gen_random_uuid()
 tenant_id   | uuid                     |           | not null | 
 source_type | text                     |           | not null | 
 source_id   | uuid                     |           | not null | 
 invoice_id  | uuid                     |           | not null | 
 link_kind   | text                     |           | not null | 
 amount      | numeric(12,2)            |           |          | 
 created_at  | timestamp with time zone |           | not null | now()
 created_by  | uuid                     |           |          | 
Indexes:
    "billing_links_pkey" PRIMARY KEY, btree (id)
    "idx_billing_links_invoice" btree (invoice_id)
    "idx_billing_links_tenant" btree (tenant_id)
    "idx_billing_links_tenant_source" btree (tenant_id, source_type, source_id)
Check constraints:
    "billing_links_link_kind_check" CHECK (link_kind = ANY (ARRAY['deposit'::text, 'final'::text, 'refund'::text, 'credit_note'::text]))
Foreign-key constraints:
    "billing_links_created_by_fkey" FOREIGN KEY (created_by) REFERENCES profiles(id)
    "billing_links_invoice_id_fkey" FOREIGN KEY (invoice_id) REFERENCES invoices(id) ON DELETE CASCADE
    "billing_links_tenant_id_fkey" FOREIGN KEY (tenant_id) REFERENCES tenants(id) ON DELETE CASCADE
Policies:
    POLICY "Active members can insert billing links" FOR INSERT
      WITH CHECK (is_active_tenant_member(auth.uid(), tenant_id))
    POLICY "Active members can view billing links" FOR SELECT
      USING (is_active_tenant_member(auth.uid(), tenant_id))
